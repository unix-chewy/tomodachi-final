Custom Image Loading & Preprocessing

Loading external images and preparing them for model prediction.


from PIL import Image

image_filenames = ['/content/airplane.jpg', '/content/bird.jpg', '/content/frog.jpg']
new_images = []
for filename in image_filenames:
    img = Image.open(filename)
    img_resized = img.resize((32, 32))
    img_array = tf.keras.preprocessing.image.img_to_array(img_resized)
    img_array = img_array / 255.0
    img_array = np.expand_dims(img_array, axis=0)
    preprocessed_images.append(img_array)

Batch Prediction & Confidence Display

Running predictions and displaying class names with confidence percentages.


batch_images = np.vstack(preprocessed_images)
predictions = model.predict(batch_images)

for i, pred in enumerate(predictions):
    predicted_class_index = np.argmax(pred)
    predicted_class_name = class_names[predicted_class_index]
    confidence = np.max(pred) * 100
    print(f"Pred: {predicted_class_name}\nConf: {confidence:.2f}%")
